<script>
function showSection(id){
  document.querySelectorAll("section").forEach(s=>s.classList.remove("active"));
  document.getElementById(id).classList.add("active");
  if(id==='transliterator') document.getElementById('romanText').focus();
}

/* ---------- Backend Integration ---------- */
let autoConvert = false;

async function convertSindhi() {
  const roman = document.getElementById('romanText').value.trim();
  if(!roman){ alert("Please type some text!"); return; }

  try {
    const res = await fetch("http://127.0.0.1:8000/convert", {
      method: "POST",
      headers: {"Content-Type":"application/json"},
      body: JSON.stringify({ text: roman })
    });
    const data = await res.json();
    document.getElementById('persoText').value = data.result;
    document.getElementById('preview').innerHTML = "<b>Preview:</b> " + data.result;
  } catch (err) {
    alert("❌ Backend not running or connection failed!");
  }
}

/* ---------- Auto Convert ---------- */
function toggleAuto(){
  autoConvert = !autoConvert;
  document.getElementById('autoBtn').innerText = autoConvert ? "Auto: On" : "Auto: Off";
  if(autoConvert) convertSindhi();
}

/* ---------- Clear Function ---------- */
function clearText(){
  document.getElementById('romanText').value = "";
  document.getElementById('persoText').value = "";
  document.getElementById('preview').innerHTML = "<b>Preview:</b> <em>(Converted text appears here)</em>";
}

/* ---------- Auto Convert on Input ---------- */
document.getElementById('romanText').addEventListener('input', () => {
  if(autoConvert) convertSindhi();
});
</script>
